def generate_all_magic_numbers():
    magic = [0]
    parameters = [
        (0, 2), (2, 4), (4, 6), (2, 6), 
        (6, 10), (8, 12), (10, 14), (12, 16)
    ]
    
    for c_start, c_high_j in parameters:
        M_n = magic[-1]
        delta_n = c_start * (c_start + 2) // 4
        C_total = delta_n + c_high_j
        M_next = M_n + C_total
        magic.append(M_next)
    
    return magic

def verify_magic_number(target):
    magic = [0]
    parameters = [
        (0, 2), (2, 4), (4, 6), (2, 6), 
        (6, 10), (8, 12), (10, 14), (12, 16)
    ]
    
    for c_start, c_high_j in parameters:
        M_n = magic[-1]
        M_next = M_n + (c_start * (c_start + 2) // 4) + c_high_j
        magic.append(M_next)
        if M_next == target:
            return True, magic
    
    return False, magic

is_magic, sequence = verify_magic_number(126)
print(f"Is 126 magic? {is_magic}")
print(f"Sequence: {sequence}")
# Output: Is 126 magic? True
#         Sequence: [0, 2, 8, 20, 28, 50, 82, 126]
